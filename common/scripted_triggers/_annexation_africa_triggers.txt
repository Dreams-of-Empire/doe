  OR = {
    tag = ENG
    tag = FRA
    tag = ITA
    AND = {
      tag = GER
      NOT = { has_government = valkist }
    }
  }
  has_socialist_government = no
}

should_release_algeria_first = {
	OR = {
		country_exists = ALG
		has_country_flag = ALG_CONQ
		459 = { is_valid_annexation_state = no }
	}
}
