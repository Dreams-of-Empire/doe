	id = jap_foreign.13
	title = jap_foreign.13.t
	desc = jap_foreign.13.d 

	is_triggered_only = yes 

	immediate = {
		hidden_effect = {
			JAP = {
				set_temp_variable = { temp = harbin_railroad }
				add_to_temp_variable = { temp = harbin_currency }
				add_to_temp_variable = { temp = harbin_trade }
				multiply_temp_variable = { temp = harbin_topic }

				set_variable = { harbin_vote = temp }
				
				set_variable = { harbin_quorum = harbin_topic }
				multiply_variable = { harbin_quorum = 0.6 }

				divide_variable = { harbin_vote = var:global.harbin_conference_joined }

				clear_variable = harbin_railroad 
				clear_variable = harbin_currency
				clear_variable = harbin_trade 
				clear_variable = harbin_topic 
				if = {
					limit = {
						check_variable = {
							var = harbin_vote 
							value = harbin_quorum 
							compare = greater_than_or_equals
						}
					}
					set_country_flag = JAP_harbin_conference_ended
				}
				else_if = {
					limit = {
						check_variable = {
							var = harbin_vote 
							value = harbin_quorum 
							compare = less_than
						}
					}
					set_country_flag = JAP_harbin_conference_failed
				}
				else_if = {
					limit = {
						check_variable = { harbin_conference = 1 }
					}
					set_country_flag = JAP_harbin_conference_failed
				}
				clear_variable = harbin_conference
				clear_variable = harbin_vote
				clear_variable = harbin_quorum 
			}
		}
	}

	option = {
		name = jap_foreign.13.a
		add_stability = 0.01
		add_political_power = 10
	}
