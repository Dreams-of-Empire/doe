  desc = annexation_IND_CONQ_DESC

  immediate = {
    hidden_effect = {
      add_to_temp_array = { core_removal_array = IND }
      remove_all_cores_and_claims_of_tags = yes
    }
  }

  option = { # Restore the Raj
    name = annex.3513.a
    ai_chance = {
      base = 100
    }
    trigger = {
      OR = {
        tag = ENG
        is_ally_with = ENG
      }
      NOT = { country_exists = RAJ }
    }
    RAJ = { get_current_government_type = yes }
    set_temp_variable = { indian_tag_for_release = RAJ }
    core_india_for_faction = yes

    release_targeted_tag = yes

      restore_previous_government_type = yes

    }
  }

  option = { # Puppet India
    name = annex.3513.b
    ai_chance = {
      base = 100
    }
    trigger = {
      NOT = {
        tag = ENG
        is_ally_with = ENG
      }
    }
    if = {
      limit = { has_socialist_government = no }
      set_temp_variable = { indian_tag_for_release = IND }
      core_india_for_faction = yes

      set_temp_variable = { tag_to_release = IND }
      release_targeted_tag = yes
    } else = {
      set_temp_variable = { indian_tag_for_release = INC }
      core_india_for_faction = yes

      set_temp_variable = { tag_to_release = INC }
      release_targeted_tag = yes
    }
  }

  option = { # Fragment
    name = annex.3513.c
    ai_chance = {
      base = 100
    }
    trigger = {
      NOT = {
        tag = ENG
        is_ally_with = ENG
      }
    }
    hidden_effect = {
      IND = {
        add_state_core = 428
        add_state_core = 439
        add_state_core = 908
        add_state_core = 438
        add_state_core = 987
        add_state_core = 986
        add_state_core = 988
        add_state_core = 972
      }
    }
    set_temp_variable = { tag_to_release = IND }
    release_targeted_tag = yes
    recheck_annexations_immediate = yes
  }

  option = { # Transfer
    name = annex.3513.d
    trigger = {
      OR = {
        tag = ENG
        is_ally_with = ENG
      }
      country_exists = RAJ
      NOT = { tag = RAJ }
    }
    ai_chance = {
      base = 100
    }
    set_temp_variable = { indian_tag_for_release = RAJ }
    core_india_for_faction = yes
    every_owned_state = {
      limit = { is_core_of = RAJ }
      ROOT = {
        transfer_state = PREV
      }
    }
  }

  option = { # Integrate the territories
    name = annex.3514.e
    trigger = {
      tag = RAJ
    }
    set_temp_variable = { indian_tag_for_release = RAJ }
