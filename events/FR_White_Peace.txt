country_event = { # Serbia losing - Peace with Bulgaria?
	id = white.1
	immediate = {log = "[GetDateText]: [Root.GetName]: event white.1"}
	title = white.1.t
	desc = white.1.d
	picture = GFX_report_event_generic_battle

	trigger = {
		original_tag = SER
		SER = {
			has_war_with = BUL
			is_puppet = no # Can't be a subject
			OR = {
				is_faction_leader = yes # Must be faction leader
				is_in_faction = no # Or not be in a faction
			}
			OR = { # Must own some of the disputed states
				owns_state = 942
				owns_state = 106
				owns_state = 787
			}
		}
		BUL = {
			is_puppet = no # Can't be a subject
			OR = {
				is_faction_leader = yes # Must be faction leader
				is_in_faction = no # Or not be in a faction
			}
			OR = { # Must be occupying key regions
				controls_state = 106
				controls_state = 108
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	fire_only_once = yes

	option = {
		name = white.1.a
		ai_chance = {
			factor = 40
		}
		BUL = {
			country_event = white.2
		}
	}
	option = {
		name = white.1.b
		ai_chance = {
			factor = 0
		}
		add_political_power = -50
		add_war_support = -0.05
		add_stability = -0.1
	}
}

					owns_state = 106

			BUL = {
				transfer_state = 106
			}

			limit = {				
				SER = {
					owns_state = 787
				}
			}
			BUL = {
				transfer_state = 787
			}
		}

		every_state = {					# Is this too much?
			limit = {
				is_core_of = BUL 		# Core of winner
				is_controlled_by = BUL  # Is occupied by winner
				is_owned_by = SER		# Is owned by loser
				NOT = {					# But is not a core of the loser

				}
			}
			BUL = {
				transfer_state = PREV	# Transfer the winners uncontested cores from loser
			}
		}
	}
	option = {
		name = white.2.b
		ai_chance = {
			factor = 0
		}
		SER = {
			country_event = white.3
		}
