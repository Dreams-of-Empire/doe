country_event = { #1937 Election
	id = southafrica.15
	immediate = {log = "[GetDateText]: [Root.GetName]: event southafrica.15"}
	title = southafrica.15.t
	desc = southafrica.15.d

	picture = report_event_generic_conference

	is_triggered_only = yes

	option = {
		name = southafrica.15.a
		hidden_effect = {
			if = {
				limit = {
					check_variable = { saf_anc > saf_united }
					check_variable = { saf_anc > saf_pnp }
				}
				country_event = { days = 1 id = southafrica.16}
			}
			if = {
				limit = {
					check_variable = { saf_united > saf_anc }
					check_variable = { saf_united > saf_pnp }
				}
				country_event = { days = 1 id = southafrica.17}
			}
			if = {
				limit = {
					check_variable = { saf_pnp > saf_anc }
					check_variable = { saf_pnp > saf_united }
				}
				country_event = { days = 1 id = southafrica.18}
			}
			if = {
				limit = {
					check_variable = { saf_pnp = saf_anc }
