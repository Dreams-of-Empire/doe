					}
				}
				add_core_of = SPA
			}
			every_state = {
				limit = {
					is_core_of = SPM
					is_owned_by = SPA
					NOT = {
						is_core_of = SPA
					}
				}
				add_core_of = SPA
			}
			every_state = {
				limit = {
					is_core_of = SPF
					is_owned_by = SPA
					NOT = {
						is_core_of = SPA
					}
				}
				add_core_of = SPA
			}
			remove_ideas = SPR_SPM_SPF_SPA_spanish_civil_war_idea
			remove_ideas = SPR_surrounded_at_all_sides_idea
			remove_ideas = SPR_civil_war_offensive_planning_idea
